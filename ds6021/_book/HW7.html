12 Question 3

set.seed(7)
k <- 10
folds <- sample(1:k, nrow(Wage), replace = TRUE)

mse_lin <- numeric(k)
mse_spline <- numeric(k)

for(i in 1:k) {
  train_data <- Wage[folds != i, ]
  test_data  <- Wage[folds == i, ]
  
  m_lin <- lm(wage ~ age + education + race, data = train_data)
  m_spl <- lm(wage ~ rcs(age, 4) + education + race, data = train_data)
  
  pred_lin <- predict(m_lin, newdata = test_data)
  pred_spl <- predict(m_spl, newdata = test_data)
  
  mse_lin[i] <- mean((test_data$wage - pred_lin)^2)
  mse_spline[i] <- mean((test_data$wage - pred_spl)^2)
}

cat("linear model:", mean(mse_lin), "\n")
linear model: 1287.18 
cat("spline model:", mean(mse_spline), "\n")
spline model: 1239.019

Yes, based on the mse values, the spline model has a lower mean squared error than the linear model.
