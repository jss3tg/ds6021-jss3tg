set.seed(7)
K <- 10
n <- nrow(d1)

fold_id <- sample(rep(1:K, length.out = n))
cv_mse_shared <- rep(NA, K)
cv_mse_offset <- rep(NA, K)
cv_mse_int <- rep(NA, K)
cv_mse_sep <- rep(NA, K)

for (k in 1:K) {
  train <- d1[fold_id != k, ]
  test  <- d1[fold_id == k, ]
  
  fit_shared <- ols(Weight ~ rcs(Height, height_knots), data = train)
  fit_offset <- ols(Weight ~ rcs(Height, height_knots) + Race1, data = train)
  fit_int <- ols(Weight ~ rcs(Height, height_knots) * Race1, data = train)
  
  cv_mse_shared[k] <- mean((predict(fit_shared, test) - test$Weight)^2)
  cv_mse_offset[k] <- mean((predict(fit_offset, test) - test$Weight)^2)
  cv_mse_int[k] <- mean((predict(fit_int, test) - test$Weight)^2)
  
  sep_preds <- rep(NA, nrow(test))
  for (r in levels(train$Race1)) {
    train_g <- train[train$Race1 == r, ]
    test_g <- test[test$Race1 == r, ]
    if (nrow(test_g) > 0) {
      fit_sep <- ols(Weight ~ rcs(Height, height_knots), data = train_g)
      sep_preds[test$Race1 == r] <- predict(fit_sep, test_g)
    }
  }
  cv_mse_sep[k] <- mean((sep_preds - test$Weight)^2)
}

data.frame(model = c("Fully shared", "Shared shape", 
                     "Full interaction", "Fully separate"),
           CV_MSE = c(mean(cv_mse_shared), mean(cv_mse_offset), 
                      mean(cv_mse_int), mean(cv_mse_sep))) |>

d_reg <- d1
contrasts(d_reg$Race1) <- contr.sum(nlevels(d_reg$Race1))
form_reg <- Weight ~ rcs(Height, height_knots) * Race1
X_full <- model.matrix(form_reg, data = d_reg)
y <- d_reg$Weight

term_labels <- attr(terms(form_reg), "term.labels")
assign_vec  <- attr(X_full, "assign")
dev_terms   <- grep("Race1", term_labels)
pf_full     <- ifelse(assign_vec %in% dev_terms, 1, 0)
X_glmnet    <- X_full[, -1]
pf          <- pf_full[-1]
set.seed(7)
cv_ridge <- cv.glmnet(X_glmnet, y, alpha = 0, penalty.factor = pf, nfolds = 10)
cv_lasso <- cv.glmnet(X_glmnet, y, alpha = 1, penalty.factor = pf, nfolds = 10)

beta_lasso_min <- coef(cv_lasso, s = "lambda.min")[-1, 1]
race_levels    <- levels(d_reg$Race1)
race_main_idx  <- which(assign_vec[-1] == dev_terms[1])
race_int_idx   <- which(assign_vec[-1] == dev_terms[2])

dev_mat <- matrix(NA_real_, nrow = length(race_levels), ncol = 4,
                  dimnames = list(race_levels, c("main", "height", "height'", "height''")))
for (j in 1:4) {
  cols <- c(race_main_idx[j], race_int_idx[(3 * (j - 1) + 1):(3 * j)])
  dev_mat[j, ] <- beta_lasso_min[cols]
}
dev_mat[5, ] <- -colSums(dev_mat[1:4, , drop = FALSE])
n_nonzero <- rowSums(dev_mat != 0)
status <- ifelse(n_nonzero == 0, "Fully pooled",
          ifelse(n_nonzero == 4, "Not pooled", "Partially pooled"))

data.frame(Race1 = race_levels, n = as.numeric(table(d_reg$Race1)[race_levels]),
           `Nonzero deviation coefficients (of 4)` = n_nonzero, Status = status, check.names = FALSE) |>

Justify using both predictive evidence (CV-MSE across the ladder and the regularized fit) and interpretive evidence (which groups’ coefficients/curves shrank the most, and whether that tracked group size, distinctiveness of the group’s curve, or something else — be honest if the pattern doesn’t look clean).

Based on our data, the CV-MSE was lowest for the Fully separate and Full interaction models. In our interpretive evidence, we can identify that the Interactiv-in-group model had a lowers MSE. This tells us that the groups have variation among the different races. We showed this by fitting our model on the lasso model and we can see that although all parameters are partially pooled, we can still find distinction between the races. We can identify that the Fully separate or Full interaction model show that the penalty for these models provides the best fit for the data and its parameters.
